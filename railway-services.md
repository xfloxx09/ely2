# Railway service configs - set these in Railway dashboard

# WEB SERVICE
# Build: pnpm install && pnpm --filter @ely/web build
# Start: pnpm --filter @ely/web start
# Port: 3000

# API SERVICE  
# Build: pnpm install
# Start: pnpm --filter @ely/api dev (or tsx apps/api/src/index.ts)
# Port: 3001

# WS SERVICE
# Build: pnpm install
# Start: pnpm --filter @ely/ws dev
# Port: 3002

# WORKER SERVICE
# Build: pnpm install
# Start: pnpm --filter @ely/worker dev
